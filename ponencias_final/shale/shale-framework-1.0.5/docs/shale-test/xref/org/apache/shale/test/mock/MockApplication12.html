159     /*** {@inheritDoc} */
160     public UIComponent createComponent(ValueExpression expression,
161                                        FacesContext context,
162                                        String componentType) {
163 
164         UIComponent component = null;
165         try {
166             component = (UIComponent) expression.getValue(context.getELContext());
167             if (component == null) {
168                 component = createComponent(componentType);
169                 expression.setValue(context.getELContext(), component);
170             }
171 
172         } catch (Exception e) {
173             throw new FacesException(e);
174         }
175         return component;
176 
177     }

201     /*** {@inheritDoc} */
202     public ELResolver getELResolver() {
203 
204         if (resolver == null) {
205 
206             // Configure a default ELResolver per Section 5.6.2 of JSF 1.2
207             CompositeELResolver composite = new CompositeELResolver();
208 
209             composite.add(new FacesImplicitObjectELResolver());
210 
211             CompositeELResolver nested = new CompositeELResolver();
212             // FIXME - nested.add() "ELResolvers from application configuration resources"
213             nested.add(new FacesVariableResolverChainWrapper());
214             nested.add(new FacesPropertyResolverChainWrapper());
215             Iterator items = resolvers.iterator();
216             while (items.hasNext()) {
217                 nested.add((ELResolver) items.next());
218             }
219             composite.add(nested);
220 
221             // composite.add(new faces.ManagedBeanELResolver()); // FIXME
222             composite.add(new ResourceBundleELResolver());
223             composite.add(new FacesResourceBundleELResolver());
224             composite.add(new MapELResolver());
225             composite.add(new ListELResolver());
226             composite.add(new ArrayELResolver());
227             composite.add(new BeanELResolver());
228             composite.add(new FacesScopedAttributeELResolver());
229 
230             // Make the resolver we have configured the application wide one
231             resolver = composite;
232 
233         }
234         return resolver;
235 
236     }

247     /*** {@inheritDoc} */
248     public ResourceBundle getResourceBundle(FacesContext context, String name) {
249 
250         if ((context == null) || (name == null)) {
251             throw new NullPointerException();
252         }
253         Locale locale = null;
254         UIViewRoot viewRoot = context.getViewRoot();
255         if (viewRoot != null) {
256             locale = viewRoot.getLocale();
257         }
258         if (locale == null) {
259             locale = Locale.getDefault();
260         }
261         return ResourceBundle.getBundle(name, locale);
262 
263     }
