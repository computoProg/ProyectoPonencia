252     private UIComponent findComponent(UIComponent parent, String id) {
253         if (parent == null) {
254             return null;
255         }
256 
257         if (parent.getId() != null && parent.getId().equals(id)) {
258             return parent;
259         } else {
260             Iterator ci = parent.getChildren().iterator();
261             while (ci.hasNext()) {
262                 UIComponent child = (UIComponent) ci.next();
263                 UIComponent target = findComponent(child, id);
264                 if (target != null) {
265                     return target;
266                 }
267             }
268         }
269 
270         return null;
271     }
