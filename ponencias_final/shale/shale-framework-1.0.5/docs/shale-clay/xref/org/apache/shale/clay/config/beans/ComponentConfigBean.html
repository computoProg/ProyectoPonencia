142      * <p>Initialization method that is passed the <code>ServletContext</code>
143      * as a parameter.  Loads the <code>sufixes</code> for the ServletContext
144      * initialization parameter.
145      * </p>

147      * @param context servlet context

149     public  void init(ServletContext context) {
150         this.context = context;
151 
152         if (suffixes == null) {
153             suffixes = new String[2];
154 
155             suffixes[0] = context.getInitParameter(
156                     Globals.CLAY_HTML_TEMPLATE_SUFFIX);
157             if (suffixes[0] == null) {
158                 suffixes[0] = Globals.CLAY_DEFAULT_HTML_TEMPLATE_SUFFIX;
159             }
160 
161             suffixes[1] = context.getInitParameter(
162                     Globals.CLAY_XML_TEMPLATE_SUFFIX);
163             if (suffixes[1] == null) {
164                 suffixes[1] = Globals.CLAY_DEFAULT_XML_TEMPLATE_SUFFIX;
165             }
166 
167         }
168 
169         String autoReloadClayFiles = context.getInitParameter(Globals.AUTO_RELOAD_CLAY_FILES);
170         if (autoReloadClayFiles != null) {
171             try {
172                 isWatchDogOn = Boolean.valueOf(autoReloadClayFiles).booleanValue();
173             } catch (RuntimeException e) {
174                 isWatchDogOn = false;
175             }
176 
177         }
178 
179         //loads the config files
180         loadConfigFiles();
181     }

184     /***
185      * <p>Loads the {@link org.apache.shale.clay.component.Clay} configration files
186      * into the <code>displayElements</code> Map.  The files are defined by the
187      * <code>clay-template-suffix</code> initialization parameter in the web deployment
188      * descriptor.  The default configuration file "META-INF/view-config.xml" is always
189      * loaded from the shale-clay java archive.</p>
190      */
191     protected void loadConfigFiles() {
192 
193         parser = new ClayXmlParser();
194         parser.setConfig(this);
195 
196         // grab the default config file
197         StringBuffer configFiles = new StringBuffer(
198                 Globals.DEFAULT_CLAY_CONFIG_FILE);
199 
200         // a comma delimited value list of config files
201         String param = context.getInitParameter(Globals.CLAY_CONFIG_FILES);
202 
203         // add the default config file
204         if (param != null && param.trim().length() > 0) {
205             configFiles.append(", ").append(param);
206 
207             log.warn(messages.getMessage("config.deprecated.param",
208                      new Object[] {Globals.CLAY_CONFIG_FILES,
209                                    Globals.CLAY_COMMON_CONFIG_FILES}));
210         }
211 
212         // a comma delimited value list of config files
213         param = context.getInitParameter(Globals.CLAY_COMMON_CONFIG_FILES);
214         if (param != null && param.trim().length() > 0) {
215             configFiles.append(", ").append(param);
216         }
217 
218         // pass the config bean to the parser
219         parser.setConfig(this);
220 
221         // map holding the resource watchers
222         watchDogs = Collections.synchronizedMap(new TreeMap());
223 
224         // create the watch dog with a list of config files to look for changes
225         WatchDog watchDog = new WatchDog(getConfigDefinitions(configFiles.toString()),
226                 Globals.DEFAULT_COMPONENT_CONFIG_WATCHDOG);
227 
228         // adds the watcher to a map identified by name
229         watchDogs.put(watchDog.getName(), watchDog);
230 
231         // loads the config files
232         watchDog.refresh(true);
233 
234         param = null;
235         configFiles = null;
236 
237     }

239     /***
240      * <p>Passed a comma delimited list of configuration files, this method returns
241      * an array of {@link ConfigBean.ConfigDefinition} defining the files.</p>
242      *
243      * @param configFiles comma seperated list of config files
244      * @return config definitions for the files
245      */
246     protected ConfigBean.ConfigDefinition[] getConfigDefinitions(String configFiles) {
247 
248         List urls = new ArrayList();
249 
250         ClassLoader classloader = Thread.currentThread().getContextClassLoader();
251         if (classloader == null) {
252             classloader = this.getClass().getClassLoader();
253         }
254 
255         // convert a tokenized list of configuration files into an array of urls
256         StringTokenizer tokenizer = new StringTokenizer(configFiles, ", ");
257         while (tokenizer.hasMoreTokens()) {
258             StringBuffer configFile = new StringBuffer(tokenizer.nextToken().trim());
259 
260             //look for a classpath prefix.
261             int i = configFile.indexOf(Globals.CLASSPATH_PREFIX);
262             if (i > -1) {
263                configFile.delete(0, i + Globals.CLASSPATH_PREFIX.length());
264             }
265 
266             try {
267                 if (i > -1) {
268                    for (Enumeration ui = classloader.getResources(configFile.toString());
269                          ui.hasMoreElements();) {
270                        urls.add(ui.nextElement());
271                    }
272                 } else {
273                    URL url = context.getResource(configFile.toString());
274                    if (url == null) {
275                       throw new PageNotFoundException(messages.getMessage("file.notfound",
276                               new Object[] {configFile.toString()}), configFile.toString());
277                    }
278                    urls.add(url);
279                 }
280             } catch (IOException e) {
281                 log.error(e);
282             }
283 
284             configFile = null;
285         }
286         tokenizer = null;
287         classloader = null;
288 
289         ConfigBean.ConfigDefinition[] configDefs = new ConfigBean.ConfigDefinition[urls.size()];
290 
291         for (int i = 0; i < urls.size(); i++) {
292             configDefs[i] = new XmlConfigDef((URL) urls.get(i));
293         }
294 
295         return configDefs;
296     }

352     /***
353      * <p>This method is called on startup to resolve the meta inheritance relationships for
354      * each top-level components in the <code>displayElements</code> collection.  There are
355      * three steps, find parents, check for circular relationships, and realize the relationships.
356      * </p>
357      */
358     public void resolveInheritance() {
359 
360         if (log.isInfoEnabled()) {
361             log.info(messages.getMessage("resolve.inheritance.begin"));
362         }
363 
364         // fixup heritage links
365         if (log.isInfoEnabled()) {
366             log.info(messages.getMessage("finding.parents"));
367         }
368 
369         Iterator di = displayElements.entrySet().iterator();
370         while (di.hasNext()) {
371             Map.Entry e = (Map.Entry) di.next();
372             ComponentBean b = (ComponentBean) e.getValue();
373 
374             try {
375                 assignParent(b);
376             } catch (RuntimeException e1) {
377                 log.error(messages.getMessage("finding.parents.exception"), e1);
378                 throw e1;
379             }
380 
381             b = null;
382             e = null;
383         }
384         di = null;
385 
386         if (log.isInfoEnabled()) {
387             log.info(messages.getMessage("checking.inheritance"));
388         }
389 
390         // Check for circular inheritance
391         di = displayElements.entrySet().iterator();
392         while (di.hasNext()) {
393             Map.Entry e = (Map.Entry) di.next();
394             ComponentBean b = (ComponentBean) e.getValue();
395 
396             try {
397                 checkCircularInheritance(b);
398             } catch (RuntimeException e1) {
399                 log.error(
400                         messages.getMessage("checking.inheritance.exception"),
401                         e1);
402                 throw e1;
403             }
404 
405             b = null;
406             e = null;
407         }
408         di = null;
409 
410         if (log.isInfoEnabled()) {
411             log.info(messages.getMessage("realizing.inheritance"));
412         }
413 
414         // now realizing inheritance
415         di = displayElements.entrySet().iterator();
416         while (di.hasNext()) {
417             Map.Entry e = (Map.Entry) di.next();
418             ComponentBean b = (ComponentBean) e.getValue();
419 
420             try {
421                 realizingInheritance(b);
422             } catch (RuntimeException e1) {
423                 log.error(messages.getMessage("realizing.inheritance.exception"), e1);
424                 throw e1;
425             }
426 
427             //check to make sure that there is not a duplicate component id
428             //within the same naming container.
429             checkTree(b);
430 
431             b = null;
432             e = null;
433         }
434         di = null;
435 
436         if (log.isInfoEnabled()) {
437             log.info(messages.getMessage("resolve.inheritance.end"));
438         }
439 
440     }

463     /***
464      * <p>Called to assign the IsA parent to the {@link ComponentBean}
465      * using the <code>extends</code> attribute.
466      * </p>
467      *
468      * @param b component bean needing isa parent assigned
469      */
470     public synchronized void assignParent(ComponentBean b) {
471 
472 
473         if (b instanceof InnerComponentBean) {
474 
475             // these elements are like inner classes
476             // set the extends to the element so the
477             // parent can be generically resolved
478             if (b.getJsfid() != null) {
479                 b.setExtends(b.getJsfid());
480             }
481         }
482 
483         // look for a meta inheritance property
484         if (b.getExtends() != null) {
485 
486             // assign the parent to a top-level display element
487             b.setIsAParent(getTopLevelElement(b.getExtends()));
488         }
489 
490         // resolve inheritance of nested components
491         Iterator ci = b.getChildrenIterator();
492         while (ci.hasNext()) {
493             assignParent((ComponentBean) ci.next());
494         }
495 
496         // resolve inheritance of converter
497         if (b.getConverter() != null) {
498             assignParent(b.getConverter());
499         }
500 
501         // resolve inheritance of validators
502         Iterator vi = b.getValidatorIterator();
503         while (vi.hasNext()) {
504             assignParent((ComponentBean) vi.next());
505         }
506 
507         // resolve inheritance of value change listeners
508         vi = b.getValueChangeListenerIterator();
509         while (vi.hasNext()) {
510             assignParent((ComponentBean) vi.next());
511         }
512         vi = null;
513 
514         // resolve inheritance of action listeners
515         vi = b.getActionListenerIterator();
516         while (vi.hasNext()) {
517             assignParent((ComponentBean) vi.next());
518         }
519         vi = null;
520 
521 
522     }

524     /***
525      * <p>This overload handles fixing up {@link AttributeBean}
526      * inheritance.</p>
527      *
528      * @param a attribute needing inheritance resolved
529      */
530     protected void realizingInheritance(AttributeBean a) {
531 
532         // look to see if the inheritance has
533         // already been resolved
534         if (a.isInheritanceFinal()) {
535             return;
536         }
537 
538         // if no parent, nothing to do
539         if (a.getIsAParent() == null) {
540             return;
541         }
542 
543         // traverse up to the parent and work down
544         realizingInheritance(a.getIsAParent());
545 
546         // inherit attribute value
547         if (a.getValue() == null) {
548             a.setValue(a.getIsAParent().getValue());
549         }
550 
551         // inherit late binding type
552         if (a.getBindingType() == null) {
553             a.setBindingType(a.getIsAParent().getBindingType());
554         }
555 
556         if (a.getDescription() == null) {
557             a.setDescription(a.getIsAParent().getDescription());
558         }
559 
560         // set final indicator
561         a.setInheritanceFinal(true);
562 
563     }

565     /***
566      * <p>This method is passed a {@link ComponentBean} and is
567      * recursively called for each contained component.  It fixes up
568      * the meta inheritance relationships.
569      * </p>
570      *
571      * @param b component bean needing inheritance realized
572      */
573     public void realizingInheritance(ComponentBean b) {
574 
575         // look at the final indicator to determine
576         // if inheritance has already been resolved
577         if (b.isInheritanceFinal()) {
578             return;
579         }
580 
581         // does the node have a parent
582         if (b.getIsAParent() != null) {
583 
584             // resolve the parents inheritance and
585             // work down
586             realizingInheritance(b.getIsAParent());
587 
588             // inherit component type
589             if (b.getComponentType() == null) {
590                 b.setComponentType(b.getIsAParent().getComponentType());
591             }
592 
593             if (b.getAllowBody() == null) {
594                 b.setAllowBody(b.getIsAParent().getAllowBody());
595             }
596 
597             if (b.getFacetName() == null) {
598                 b.setFacetName(b.getIsAParent().getFacetName());
599             }
600 
601             if (b.getDescription() == null) {
602                 b.setDescription(b.getIsAParent().getDescription());
603             }
604 
605             // inherit parents attributes
606             Iterator pi = b.getIsAParent().getAttributeIterator();
607             while (pi.hasNext()) {
608                 AttributeBean a = (AttributeBean) pi.next();
609                 if (a != null) {
610                     // if the parent has and attribute the child doesn't
611                     // then the child inherits it
612                     if ((a.getName() != null)
613                         && (!b.getAttributes().containsKey(a.getName()))) {
614                         b.getAttributes().put(a.getName(), a);
615                     } else if (
616                             (a.getName() != null)
617                             && (b.getAttributes().containsKey(a.getName()))) {
618                         // if the parent has an attribute, let the child
619                         // attribute inherit properties of the attribute
620                         AttributeBean ca =
621                                 (AttributeBean) b.getAttributes().get(a.getName());
622                         ca.setIsAParent(a);
623                         realizingInheritance(ca);
624                     }
625 
626                 }
627                 a = null;
628             }
629             pi = null;
630 
631             //inherit symbols
632             pi = b.getIsAParent().getSymbols().entrySet().iterator();
633             while (pi.hasNext()) {
634                Map.Entry e = (Map.Entry) pi.next();
635                if (!b.getSymbols().containsKey(e.getKey())) {
636                   b.getSymbols().put(e.getKey(), e.getValue());
637                }
638             }
639 
640             //inherit elements from the parent.  elements are identified/ordered by
641             //the renderid in the set.
642             TreeSet tmp = new TreeSet();
643             // get the parents children/elements
644             Iterator ci = b.getIsAParent().getChildrenIterator();
645             while (ci.hasNext()) {
646                 ComponentBean c = (ComponentBean) ci.next();
647                 // the parent of a element can only be a display element
648                 // and a display element must have or inherit a component type
649                 if (c.getComponentType() == null) {
650                     throw new NullPointerException(messages.getMessage("missing.componentType.exception",
651                             new Object[] {c}));
652                 }
653 
654                 // if the child node doesn't contain the parent's
655                 // element identified by renderId then add it to a
656                 // temp set
657                 if (!b.getChildren().contains(c)) {
658                     tmp.add(c);
659                 }
660 
661                 c = null;
662             }
663 
664             // merge the delta of parent elements not found in the
665             // child's element set into the child's element set
666             b.setChildren(tmp);
667             tmp = null;
668             ci = null;
669 
670         }
671 
672         // at this point, the component id should be defined or inherited
673         if (b.getComponentType() == null) {
674             throw new NullPointerException(messages.getMessage("missing.componentType.exception",
675                     new Object[] {b}));
676         }
677 
678         // iterate thru the inherited set of elements resolving
679         // their inheritance
680         Iterator ci = b.getChildren().iterator();
681         while (ci.hasNext()) {
682             ComponentBean c = (ComponentBean) ci.next();
683             realizingInheritance(c);
684             if (c.getComponentType() == null) {
685                 throw new NullPointerException(messages.getMessage("missing.componentType.exception",
686                         new Object[] {c}));
687             }
688             c = null;
689         }
690         ci = null;
691 
692         // inherit converter from parent
693         if (b.getConverter() == null && b.getIsAParent() != null
694                 && b.getIsAParent().getConverter() != null) {
695             b.addConverter((ConverterBean) b.getIsAParent().getConverter());
696         }
697 
698         // resolve the inheritance of a nested converter
699         if (b.getConverter() != null) {
700             realizingInheritance(b.getConverter());
701         }
702 
703         // inheritance of all parent validators
704         if (b.getIsAParent() != null) {
705            Iterator vi = b.getIsAParent().getValidatorIterator();
706            while (vi.hasNext()) {
707                ComponentBean c = (ComponentBean) vi.next();
708                // check to make sure the child doesn't have one
709                if (!b.getValidators().contains(c)) {
710                   b.addValidator((ValidatorBean) c);
711                }
712                c = null;
713            }
714            vi = null;
715         }
716 
717         // resovle inheritance of all nested validators
718         Iterator vi = b.getValidatorIterator();
719         while (vi.hasNext()) {
720             ComponentBean c = (ComponentBean) vi.next();
721             realizingInheritance(c);
722             c = null;
723         }
724         vi = null;
725 
726         // inheritance of all value change listeners
727         if (b.getIsAParent() != null) {
728            vi = b.getIsAParent().getValueChangeListenerIterator();
729            while (vi.hasNext()) {
730               ComponentBean c = (ComponentBean) vi.next();
731               if (!b.getValueChangeListeners().contains(c)) {
732                  b.addValueChangeListener((ValueChangeListenerBean) c);
733               }
734               c = null;
735            }
736            vi = null;
737         }
738 
739         // resolve inheritance of all nested value change listeners
740         vi = b.getValueChangeListenerIterator();
741         while (vi.hasNext()) {
742             ComponentBean c = (ComponentBean) vi.next();
743             realizingInheritance(c);
744             c = null;
745         }
746         vi = null;
747 
748         // inheritance of all action listeners
749         if (b.getIsAParent() != null) {
