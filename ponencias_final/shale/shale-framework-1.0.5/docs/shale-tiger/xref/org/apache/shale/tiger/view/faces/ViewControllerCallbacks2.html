60      /***
61       * <p>Perform the <code>preprocess</code> callback on the specified
62       * instance.</p>
63       *
64       * @param instance Bean instance on which to perform this callback
65       */
66      public void preprocess(Object instance) {
67  
68          if (instance instanceof ViewController) {
69              try {
70                  ((ViewController) instance).preprocess();
71              } catch (Exception e) {
72                  handleException(FacesContext.getCurrentInstance(), e);
73              }
74              return;
75          }
76  
77          Method method = method(instance, Preprocess.class);
78          if (method != null) {
79              try {
80                  method.invoke(instance, new Object[0]);
81              } catch (IllegalAccessException e) {
82                  handleException(FacesContext.getCurrentInstance(), e);
83              } catch (InvocationTargetException e) {
84                  handleException(FacesContext.getCurrentInstance(), (Exception) e.getCause());
85              }
86          }
87  
88      }

132

137         return (method(instance, Destroy.class) != null);

143     /***
144      * <p>The set of method annotations for callbacks of interest.</p>
145      */
146     private static final Class[] annotations =
147     { Preprocess.class, Prerender.class, Destroy.class };
148 
149

151     /***
152      * <p>Data structure to maintain information about annotated
153      * methods.  In this map, the key is the Class being analyzed,
154      * and the value is an inner map.  In the inner map, the key
155      * is an Annotation class, and the value is the corresponding
156      * Method instance.</p>
157      */
158     private transient Map<Class,Map<Class,Method>> maps =
159       new HashMap<Class,Map<Class,Method>>();
160 
161 
162     /***
163      * <p>Return the <code>Method</code> to be called for the specified
164      * annotation on the specified instance, if any.  If there is no such
165      * method, return <code>null</code>.</p>
166      *
167      * @param instance Instance on which callbacks will be performed
168      * @param annotation Annotation for which to return a method
169      */
170     private Method method(Object instance, Class annotation) {
171 
172         // Does the underlying class implement the View annotation?
173         // If not, exit early
174         Class clazz = instance.getClass();
175         if (clazz.getAnnotation(View.class) == null) {
176             return null;
177         }
178 
179         synchronized (maps) {
180 
181             // If we have seen this Class already, simply return the
182             // previously located Method (if any)
183             Map<Class,Method> map = maps.get(clazz);
184             if (map != null) {
185                 return map.get(annotation);
186             }
187 
188             // Construct and cache a new Map identifying the
189             // methods of interest for these callbacks
190             map = new HashMap<Class,Method>();
191             Method[] methods = clazz.getMethods();
192             for (int i = 0; i < methods.length; i++) {
193                 if (methods[i].getParameterTypes().length > 0) {
194                     continue;
195                 }
196                 for (int j = 0; j < annotations.length; j++) {
197                     if (methods[i].getAnnotation(annotations[j]) != null) {
198                         map.put(annotations[j], methods[i]);
199                     }
200                 }
201             }
202             maps.put(clazz, map);
203             return map.get(annotation);
204 
205         }
206 
207     }
