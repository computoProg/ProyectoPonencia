18  package org.apache.shale.util;
19  
20  import javax.faces.FacesException;
21  import javax.faces.component.UIViewRoot;
22  import javax.faces.context.FacesContext;
23  import javax.faces.convert.Converter;
24  import javax.faces.convert.ConverterException;
25  
26  /***
27   * <p>Helper class to provide access to by-type JavaServer Faces
28   * <code>Converter</code> capabilities.  This implementation is stateless
29   * and maintains no cached information, so instances may be freely created
30   * and destroyed with no side effects.</p>
31   *
32   * $Id: ConverterHelper.java 560569 2007-07-28 16:56:48Z gvanmatre $
33   *
34   * @since 1.0.1
35   */
36  public class ConverterHelper {
37  
38  
39      // -------------------------------------------------------- Static Variables
40  
41  
42      /***
43       * <p>Messages for this class.</p>
44       */
45      private static Messages messages =
46              new Messages("org.apache.shale.resources.Bundle",
47                           ConverterHelper.class.getClassLoader());
48  
49  
50      // ---------------------------------------------------------- Public Methods
51  
52  
53      /***
54       * <p>Use the registered by-type <code>Converter</code> to convert the
55       * specified String value to a corresponding Object value.</p>
56       *
57       * @param context <code>FacesContext</code> for the current request
58       * @param type Type to which this value should be converted
59       *  (used to select the appropriate by-type converter)
60       * @param value Value to be converted
61       *
62       * @exception ConverterException if a conversion error occurs
63       */
64      public Object asObject(FacesContext context, Class type, String value) {
65  
66          if (String.class == type) {
67              return value;
68          }
69  
70          UIViewRoot root = context.getViewRoot();
71          if (root == null) {
72              root = new UIViewRoot();
73          }
74          
75          return converter(context, type).getAsObject(context, root, value);
76  
77      }
78  
79  
80      /***
81       * <p>Use the registered by-type <code>Converter</code> to convert the
82       * specified Object value to a corresponding String value.</p>
83       *
84       * @param context <code>FacesContext</code> for the current request
85       * @param type Type from which this value should be converted
86       *  (used to select the appropriate by-type converter)
87       * @param value Value to be converted
88       *
89       * @exception ConverterException if a conversion error occurs
90       */
91      public String asString(FacesContext context, Class type, Object value) {
92  
93          if (value == null) {
94              return null;
95          } else if ((String.class == type) && (value instanceof String)) {
96              return (String) value;
97          }
98          UIViewRoot root = context.getViewRoot();
99          if (root == null) {
100             root = new UIViewRoot();
101         }
102 
103         return converter(context, type).getAsString(context, root, value);
104 
105     }
106 
107 
108     // --------------------------------------------------------- Private Methods
109 
110 
111     /***
112      * <p>Return an appropriate <code>Converter</code> instance for the
113      * specified type.</p>
114      *
115      * @param context <code>FacesContext</code> for the current request
116      * @param type Type for which to retrieve a <code>Converter</code>
117      *
118      * @exception ConverterException if no <code>Converter</code> has been
119      *  registered for the specified type
120      */
121     private Converter converter(FacesContext context, Class type) {
122 
123         if (type == null) {
124             throw new ConverterException(messages.getMessage("convHelper.missing",
125                                                              context.getViewRoot().getLocale()));
126         }
127 
128         Converter converter = null;
129         try {
130             converter = context.getApplication().createConverter(type);
131         } catch (FacesException e) {
132             throw new ConverterException(e);
133         }
134         if (converter == null) {
135             throw new ConverterException(messages.getMessage("convHelper.noConverter",
136                                                              context.getViewRoot().getLocale(),
137                                                              new Object[] { type.getName() }));
138         }
139         return converter;
140 
141     }
