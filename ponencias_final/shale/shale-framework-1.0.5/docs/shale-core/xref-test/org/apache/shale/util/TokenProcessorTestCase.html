101     // Test generation of multiple tokens and the ability to validate them
102     public void testMultiple() {
103 
104         Set set = new HashSet();
105 
106         // Create a bunch of tokens
107         for (int i = 0; i < 1000; i++) {
108             String token = null;
109             while (true) {
110                 token = tp.generate(facesContext);
111                 if (!set.contains(token)) {
112                     break;
113                 }
114             }
115             set.add(token);
116         }
117 
118         // Ensure that we can verify all of them
119         Iterator tokens = set.iterator();
120         while (tokens.hasNext()) {
121             String token = (String) tokens.next();
122             assertTrue("Cannot verify token '" + token + "'", tp.verify(facesContext, token));
123         }
124         
125     }

308     //looks for an html attribute in a markup snippet
309     private String getAttribute(StringBuffer htmlSnippet, String attributeName) {
310         String sarg = attributeName + "=\"";
311         int s = htmlSnippet.indexOf(sarg);
312         String value = null;
313         if (s > -1) {
314            s += sarg.length();
315            int e = htmlSnippet.indexOf("\"", s);
316            if (e > -1 && e > s) {
317               value = htmlSnippet.substring(s, e);
318            }
319            
320         }
321         return value;
322     }

324     //looks for an error message added in validation 
325     private void checkMessage(String expectedSummary, String expectedDetail, UIInput component) {
326         String id = component.getClientId(facesContext);
327         Iterator mi = facesContext.getMessages(id);
328         boolean hasMessage = false;
329         while (mi.hasNext()) {
330            FacesMessage message = (FacesMessage) mi.next();    
331            String summary = message.getSummary();
332            String detail = message.getDetail();
333            assertEquals(expectedSummary, summary);
334            assertEquals(expectedDetail, detail);
335            hasMessage = true;
336         }
337         
338         assertTrue(id + " has messages", hasMessage);   
339     }

341     //checks to make sure there are no error messages after validation
342     private void checkNoMessages(UIInput component) {
343         String id = component.getClientId(facesContext);
344         Iterator mi = facesContext.getMessages(id);
345         boolean hasMessage = false;
346         while (mi.hasNext()) {
347            hasMessage = true;
348         }
349         
350         assertFalse(id + " has messages", hasMessage);   
351        
352     }

355     //render and return the results
356     private StringBuffer encode(UIComponent component) throws Exception {
357         //builds a buffer to write the page to
358         StringWriter writer = new StringWriter();
359         //create a buffered response writer
360         ResponseWriter buffResponsewriter = facesContext.getRenderKit()
361                         .createResponseWriter(writer, null, response.getCharacterEncoding());
362         //push buffered writer to the faces context
363         facesContext.setResponseWriter(buffResponsewriter);
364         //start a document
365         buffResponsewriter.startDocument();
366         
367         //render HTML
368         component.encodeBegin(facesContext);
369         component.encodeChildren(facesContext);
370         component.encodeEnd(facesContext);
371         
372         //end the document
373         buffResponsewriter.endDocument();
374         
375         return writer.getBuffer();
376     }
